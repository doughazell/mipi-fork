def evaluate(delimiter = ' ')
  if (attribute_name != "[ALL]") then
    @values = Array.new
    attribute_name.split(delimiter).each do |attribute_name|
      @value = data_element[attribute_name]
      if (@value.presence) then
        @values << @value 
      end
    end
    @values.join(delimiter)
  end
end
